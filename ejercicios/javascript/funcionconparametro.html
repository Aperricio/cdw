Función con parámetro

        <script>
            function minuto(tope) {
                let fecha = new Date();
                let elMinuto = fecha.getMinutes();
                if ( elMinuto < tope ) {
                    alert(elMinuto + " es menos que " + tope)
                }
                else {
                    alert(elMinuto + " es más que " + tope)
                }
            }
            minuto(1);
            minuto(30);
            minuto(59);
        </script>
