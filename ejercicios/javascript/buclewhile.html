Ejemplo de bucle con while

        <script>
            let fecha = new Date();
            while (fecha.getSeconds() < 30 ) {
                alert(fecha.getSeconds() + ", aún no pasan de 30");
                fecha = new Date() //Sin esta línea, siempre consulta la misma fecha. Si está en los primeros 30s, no sale del bucle
             }
            alert(fecha.getSeconds() + ", ya ha pasado de 30");
        </script>
