Ejemplo de sleep (function y while)

        <script>
            function sleep(miliseconds) {
                let ahora = new Date().getTime();
                while ( ahora + ( miliseconds * 1000) >= new Date().getTime()) {}
            }
            function minuto() {
                let fecha = new Date();
                alert("Estás en las: " + fecha.getHours() + ":" + fecha.getMinutes());
            }
            minuto();
            sleep(10);
            minuto();
        </script>
