Funcion y document.writeln

document.writeln es lo mismo que document.write, pero añade \n (salto de línea)

        <script>
            function miliSegundo() {
                let fecha = new Date();
                let elMSegundo = fecha.getMilliseconds();
                if (elMSegundo < 500) {
                    document.writeln("<p>" + i + " - " + elMSegundo + " menos de 500. </p>")
                }
                else {
                    document.writeln("<p>" + i + " - " + elMSegundo + " más de 500. </p>")
                }
            }
            for ( i = 0; i <= 9; i++) {
                miliSegundo()
            }
        </script>
